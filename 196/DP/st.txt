def find_all_occurrences(s, t):
    occurrences = []
    start_index = 0
    while True:
        index = s.find(t, start_index)
        if index == -1:
            break
        occurrences.append(index)
        start_index = index + 1
    return occurrences
s = input()
t = input()
results = find_all_occurrences(s, t)
print(*results)
